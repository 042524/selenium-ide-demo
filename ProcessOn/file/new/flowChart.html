新建 Flowchart 流程图的各种模板流程图测试案例
store 1 n
while ${n} < 6
open new
click link=Flowchart流程图
waitForText //div[2]/div[2]/span 跨职能流程图（垂直）
click //div[2]/div[${n}]
click id=chart_title
storeEval 'flowChart' + Math.round(Math.random()*1000) fileName
type id=chart_title ${fileName}
clickAndWait id=template_select_ok
assertTitle ${fileName} - ProcessOn
verifyText css=div.panel_container:nth-child(2) h3 Flowchart 流程图
clickAndWait css=span.ico.ico_goback
assertTitle ProcessOn - 我的文件
waitForText css=div.title 标题
verifyTextPresent ${fileName}
storeEval ${n} + 1 n
endWhile
